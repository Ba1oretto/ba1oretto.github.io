```java
    /* 静态代码块
     * 格式: static{}
     * 位置: 类里方法外
     * 执行时机: 静态代码块也属于静态资源, 随着类的加载而加载, 优先于对象加载
     * 并且静态资源只会加载一次
     * 作用: 用于加载那些需要第一时间就加载, 并且只加载一次的资源, 常用来初始化*/

    static {
        System.out.println("静态代码块");
    }

    /* 构造代码块
     * 位置: 类里方法外
     * 执行时机: 创建对象时执行, 并且优先于构造方法执行
     * 作用: 用于提取所有构造方法的共性功能*/

    {
        System.out.println("构造代码块");
    }

    public apple() {
        System.out.println("无参构造");
    }

    public void clean() {
        System.out.println("普通方法");
        /* 局部代码块
         * 位置: 方法里
         * 执行时机: 每次调用局部代码块所处的方法时才会执行
         * 作用: 用于控制变量的作用范围*/
        {

        }
    }
}
```

```java
/* 本类用于静态static的入门案例*/
public class testStatic_01 {
    public static void main(String[] args) {
        fruit f = new fruit();
        fruit f2 = new fruit();
        f.grow();
        fruit.clean();
        System.out.println(f.weight);
        System.out.println(fruit.kind);
        System.out.println("""


                """);

        /* 3.静态资源在内存中只有一份, 而且被全局所有对象共享
         * 所以: 当我们通过任意一种方式修改了静态变量的值以后
         * 不管用何种方式查看, 静态变量的值都是刚刚修改后的值*/
        fruit.kind = "梨";
        System.out.println(f.kind);
        System.out.println(f2.kind);
        System.out.println(fruit.kind);
        System.out.println("-------------");
        fruit.kind = "苹果";
        System.out.println(f.kind);
        System.out.println(f2.kind);
        System.out.println(fruit.kind);
        System.out.println("-------------");
        fruit.kind = "橘子";
        System.out.println(f.kind);
        System.out.println(f2.kind);
        System.out.println(fruit.kind);
    }
}

/* 被static修饰的资源统称为静态资源
 * 静态资源是随着类加载而加载到内存中的, 比对象优先进入内存
 * 所以静态资源可以通过类名直接调用, 即使没有对象, 也可以调用*/
class fruit {

    /* 1.可以使用static修饰属性*/
    static String kind;
    double weight;

    /* 2.可以用static修饰方法*/
    public static void clean() {
        kind = "apple";
        System.out.println("洗水果");
    }

    public void grow() {
        System.out.println("这个果子长得一看就很好吃");
    }
}
```

```java
public class testPolymorphism_01 {
    public static void main(String[] args) {
        animal a = new animal();
        cat c = new cat();
        dog d = new dog();
        a.eat();
        c.eat();
        d.eat();

        System.out.println("""
                                
                ----------------
                """);

        /* 2.父类对象不可以使用子类的特有功能*/
//        a.run();
//        a.jump();
        c.jump();
        d.run();

        System.out.println("""
                                
                ----------------
                """);

        /* 3.口诀一: 父类引用指向子类对象
         * 解释: 创建出来的子类对象的地址值, 交给父类类型的引用类型变量来保存*/
        animal a2 = new cat();
        animal a3 = new dog();

        /* 4.口诀二: 编译看左边, 运行看右边
         * 解释: 得在父类定义这个方法才能通过编译, 编译时把多态对象看作是父类类型
         * 得在子类重写这个方法才能满足多态, 实际干活的是子类*/
        a2.eat();


    }
}
/* 多态的前提: 继承+重写*/

    public void eat() {
        System.out.println("吃啥都行");
    }
}

    }

    public void jump() {

    }
```

```java
public class testPolymorphism_02 {
    public static void main(String[] args) {
        dog2 d = new dog2();
        System.out.println(d.sum);
        d.eat();

        System.out.println("""
                                
                --------------
                """);

        animal2 a = new dog2();
        /* 多态中, 成员变量使用的是父类的*/
        System.out.println(a.sum);
        /* 多态中, 方法的声明使用的是父类的, 方法体使用的是子类的*/
        a.eat();

        System.out.println("""
                                
                --------------
                """);

        /* 多态中调用的静态方法是父类的, 因为多态对象把自己看作是父类对象
         * 直接使用父类中的静态资源*/
//        a.play();
        animal2.play();
        dog2.play();

    }
}

class animal2 {
    int sum = 10;

    public void eat() {
        System.out.println("小动物吃啥都行");//方法体
    }

    public static void play() {
        System.out.println("小动物玩啥都行");
    }
}

class dog2 extends animal2 {
    int sum = 20;

    @Override
    public void eat() {
        System.out.println("小狗爱吃肉包子");
    }

    /* 这不是一个重写的方法, 只是恰巧在两个类中出现了一摸一样的两个静态方法
     * 静态方法属于类资源, 只有一份, 不存在重写
```
